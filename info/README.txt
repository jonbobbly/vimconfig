~/.vim/ Organization
====================

ftplugin/	File Type Plugins
info/		General Information
sessions/	"Session" Files
syntax/		Syntax Highlighting Files
vimrcso/	Config Files :so'ed from .vimrc

./ftplugin:
  test.vim	Beginnings of a pokedex filetype

./info:
  digraphs.txt	Listing of common digraphs
  mappings.txt	Listing of mappings
  README.txt

./sessions:
  ascii.vim	Five window layout to edit ascii files

./syntax:
  test.vim	along with ftplugin/test.vim

./vimrcso:
  functions.vim	Functions
  global.vim	Global settings
  mappings.vim	Key Mappings
  trial.vim	Testing area for new settings
